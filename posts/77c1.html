             S--代表---signed int---有符号整形

三.矩阵运算

1.矩阵加减法

cv::Mat a= Mat::eye(Size(3,2), CV_32F);//对角矩阵
//[1, 0, 0;
// 0, 1, 0]
cv::Mat b= Mat::ones(Size(3,2), CV_32F);//单位矩阵
//[1, 1, 1;
// 1, 1, 1]
cv::Mat c= a+b;//矩阵加法
//[2, 1, 1;
// 1, 2, 1]
cv::Mat d= a-b;//矩阵减法
//[0, -1, -1;
// -1, 0, -1]
cv::Mat e= a+1;//矩阵与标量相加
//[2, 1, 1;
// 1, 2, 1]
cv::Mat f= a-1;//矩阵与标量相减
//[0, -1, -1;
// -1, 0, -1]

2.矩阵点乘

cv::Mat g= a.mul(b);//矩阵点乘
// [1, 0, 0;
 // 0, 1, 0]

3.矩阵点除

cv::Mat g2= a/b;//矩阵点除
// [1, 0, 0;
// 0, 1, 0]
cv::Mat g3= a/2.0;//矩阵除以标量
//[0.5, 0, 0;
// 0, 0.5, 0]

4.矩阵乘法

cv::Mat h= Mat::ones(Size(2,3), CV_32F);//单位矩阵
//[1, 1;
// 1, 1;
// 1, 1]
cv::Mat i= h*a;//矩阵乘法 前者列数要等于后者行数 (3*2) * (2*3) =(3*3)
//[1, 1, 0;
// 1, 1, 0;
// 1, 1, 0]
cv::Mat j= h*2;//矩阵与标量相乘
//[2, 2;
// 2, 2;
// 2, 2]

5.矩阵转置

cv::Mat k= h.t();//矩阵转置
//[1, 1, 1;
// 1, 1, 1]

6.矩阵求逆

cv::Mat k= i.inv();//矩阵求逆，其中i为n*n的方阵

9.求矩阵最大值与最小值及其位置

void minMaxLoc(InputArray src, CV_OUT double* minVal,CV_OUT double* maxVal=0, CV_OUT Point* minLoc=0,CV_OUT Point* maxLoc=0, InputArray mask=noArray());
/*
参数：
src – 输入单通道矩阵（图像）.
minVal – 指向最小值的指针， 如果未指定则使用NULL
maxVal – 指向最大值的指针， 如果未指定则使用NULL
minLoc – 指向最小值位置（2维情况）的指针， 如果未指定则使用NULL
maxLoc – 指向最大值位置（2维情况）的指针， 如果未指定则使用NULL
mask – 可选的蒙版，用于选择待处理子区域
*/

示例：

// 求极值 最大、最小值及其位置
Mat img = imread("Lena.jpg",0);

imshow("original image",img);

double minVal=0,maxVal=0;//最大值、最小值

cv::Point minPt, maxPt;//对应位置

minMaxLoc(img,&minVal,&maxVal,&minPt,&maxPt);//进行计算
//输出结果
cout<<"min value = "<<minVal<<endl;

cout<<"max value = "<<maxVal<<endl;

cout<<"minPt = ("<<minPt.x<<","<<minPt.y<<")"<<endl;

cout<<"maxPt = ("<<maxPt.x<<","<<maxPt.y<<")"<<endl;

cout<<endl;
//显示在图像上
cv::Rect rectMin(minPt.x-10,minPt.y-10,20,20);

cv::Rect rectMax(maxPt.x-10,maxPt.y-10,20,20);

cv::rectangle(img,rectMin,cv::Scalar(200),2);

cv::rectangle(img,rectMax,cv::Scalar(255),2);

imshow("image with min max location",img);

cv::waitKey();

11.改变矩阵形状

(1)reshape()函数

reshape()函数只是在逻辑上改变矩阵的行列数或者通道数，没有任何的数据的复制，也不会增减任何数据，因此这是一个O(1)的操作，它要求矩阵是连续的。

Mat::reshape(int cn, int rows=0 const)
//cn：目标通道数，如果是0则保持和原通道数一致；
//rows：目标行数，同上是0则保持不变；

改变后的矩阵要满足 rows X cols X channels 跟原数组元数数量相等，所以如果原来矩阵是单通道3 X 3的，调用Reshape(0,2)是会报错的，因为3 X 3不能被2 X 1整除。

(2)resize()函数

会实际改变矩阵的行数，会引起矩阵元素在内存中的重新分配。

Mat::resize(size_t sz)
Mat::resize(size_t sz, const Scalar& s)//参数1：新形状的行数 参数2：如果sz大于原来的行数，可以设置默认填充的数值

14.对矩阵进行反转

对图像进行翻转或旋转可以使用cv::flip()函数，可以实现将一个二维矩阵沿X轴、Y轴或者同时沿XY轴翻转。函数原型如下:

C++: void flip(InputArray src, OutputArray dst, int ﬂipCode)

参数介绍：

src: 即输入矩阵

dst: 即输出矩阵

flipCode: 旋转码，即控制函数对矩阵进行怎样的旋转。

当参数flipCode=0时，将对矩阵沿X轴（垂直）方向翻转；

当flipCode=1时，将对矩阵沿Y轴（水平）方向翻转；

当flipCode=-1时，将对矩阵沿XY轴（水平、垂直）方向翻转。

下面，我们给出一段测试代码，代码中实现了将矩阵沿X轴方向旋转。

 #include <iostream>
 #include <opencv2/opencv.hpp>

 using namespace std;
 using namespace cv;

 int main()
 {
     Mat img = imread("D:/Media/Image/lena.jpg", CV_LOAD_IMAGE_ANYCOLOR);
     imshow("Input", img);

     cv::flip(img, img, 0);
     imshow("Flip", img);
     waitKey(0);

     return 0;
 }

当然，我们也可以将flipCode分别改为1（沿Y轴旋转）和-1（沿XY轴旋转）。

15.对矩阵进行拼接

在Matlab中将两个矩阵合并非常方便，按行合并，如A=[B C]。按列合并如A=[B ;C]
OpenCV中有vconcat和hconcat函数，用于矩阵的合并与图像的拼接。

vconcat（B, C, A）; // 等同于A=[B; C] C接到B的下面
hconcat（B, C, A）; // 等同于A=[B C]     C接到B的后面

五.矩阵拷贝

1.“=” 是浅拷贝

不复制数据只创建矩阵头，数据共享，指向同一块内存（更改a,b,c的任意一个都会对另外2个产生同样的作用）

Mat a = imread("1.jpg");

Mat b = a; //a "copy" to b

Mat c(a); //a "copy" to c

a.col(9)=0;//把a的第十列全置0，b和c也相应发生变化

2.“clone”是深拷贝

clone 是完全的深拷贝，在内存中申请新的空间

Mat A  = Mat::ones(4,5,CV_32F);//4行5列

Mat B = A.clone()    //clone 是完全的深拷贝，在内存中申请新的空间，与A独立

3.”copyTo()” 是深拷贝

copyTo 也是深拷贝，但是否申请新的内存空间，取决于目标矩阵头中的大小信息是否与原矩阵一致。

若一致则只深拷贝并不申请新的空间，否则先申请空间后再进行拷贝．

Mat A  = Mat::ones(4,5,CV_32F);//4行5列

Mat C;

A.copyTo(C) //此处的C矩阵大小与Ａ大小不一致，则申请新的内存空间，并完成拷贝，等同于clone()

Mat D = A.col(１);

A.col(0).copyTo(D)　//此处D矩阵大小与A.col(0)大小一致，因此不会申请空间，而是直接进行拷贝，相当于把A的第一列赋值给第二列
