int main()
{

    //避免乱码
    CPLSetConfigOption("SHAPE_ENCODING", "UTF-8");

    //注册文件格式

    //使用只读方式打开图像
    GDALDataset *poDataset = (GDALDataset*)GDALOpen(imgfile_ReadDir.c_str(), GA_ReadOnly);
    if (poDataset == NULL)
    {
        cout << "影像"<< imgfile_ReadDir <<"打开失败！" << endl;
        return 0;
    }

    //输出图像的格式信息
    printf("影像的格式信息Driver：%s/%s\n", 
        poDataset->GetDriver()->GetDescription(), 
        poDataset->GetDriver()->GetMetadataItem(GDAL_DMD_LONGNAME));


    //输出影像像元数据类型
    GDALDataType datatype = poDataset->GetRasterBand(1)->GetRasterDataType();//影像像元的数据类型，数字
    cout << "影像" << imgfile_ReadDir << "的像元数据类型为："<GetRasterXSize();    //输入影像宽
    int height = poDataset->GetRasterYSize();    //输入影像高
    int channels = poDataset->GetRasterCount();    //输入影像通道数
    cout << "影像" << imgfile_ReadDir << "的宽，高，波段数为：("<GetProjectionRef() != NULL)
    {
        cout << "影像" << imgfile_ReadDir << "的投影信息Projection is  "<GetProjectionRef() << endl;
    }
    else
    {
        cout << "影像" << imgfile_ReadDir << "不存在投影信息！" << endl;
    }

    //输出图像的坐标和分辨率信息
    double GeoTransform[6];//0和3是左上角顶点的xy坐标。1和5是像元的宽高
    if (poDataset->GetGeoTransform(GeoTransform) == CE_None)
    {
        //影像坐标系左上角顶点为坐标原点，X轴向右为正方向，Y轴向下为正方向。
        double ZeroX = GeoTransform[0];//影像左上角顶点的列坐标（经度）
        double ZeroY = GeoTransform[3];//影像左上角顶点的行坐标（纬度）
        double PixelX = GeoTransform[1];//影像分辨率（像元宽）
        double PixelY = GeoTransform[5];//影像分辨率（像元高）备注：PixelY是矢量，因Y轴向下为正方向，所以为负数
        //如果影像是指北的, GeoTransform[2] 和 GeoTransform[4] 这两个参数的值为0。

        //像素行列坐标 求 地理坐标    col:列数    row:行数
        int col = width-1, row = height-1;//右下角顶点的在原始影像中的像元行列坐标
        double GeoX = ZeroX + col * PixelX + row * GeoTransform[2];//影像右下角顶点的地理列坐标（经度）
        double GeoY = ZeroY + col * GeoTransform[4] + row * PixelY;//影像右下角顶点的地理行坐标（纬度）

        //由地理坐标 求 行列坐标。相当于已知GeoX，GeoY求row col  解二元一次方程组
        double temp = PixelX * PixelY - GeoTransform[2] * GeoTransform[4];
        col = int(((GeoX - ZeroX) * PixelY - (GeoY - ZeroY) * GeoTransform[2]) / temp); //列数
        row = int(((GeoY - ZeroY) * PixelX - (GeoX - ZeroX) * GeoTransform[4]) / temp); //行数

        cout << "影像" << imgfile_ReadDir << "左上角顶点的地理坐标X,Y为：(" << ZeroX << "," << ZeroY << ")" << endl;
        cout << "影像" << imgfile_ReadDir << "右下角顶点的地理坐标X,Y为：(" << GeoX << "," << GeoY << ")" << endl;
        cout << "影像" << imgfile_ReadDir << "的像元分辨率（像元宽，像元高）为：(" << PixelX << "," << PixelY << ")" << endl;
    }
    else

    }

    //读取一个波段
    int BandIndex = 1;//波段索引从1开始
    GDALRasterBand *poBand = poDataset->GetRasterBand(BandIndex);

    //获取该波段的最大值最小值，如果获取失败则进行统计
    int bandGotMin, bandGotMax;
    double adfMinMax[2];
    adfMinMax[0] = poBand->GetMinimum(&bandGotMin);//波段像素最小值
    adfMinMax[1] = poBand->GetMaximum(&bandGotMax);//波段像素最大值

    //获取该波段的均值、标准差

    double pdfMax;//该波段最大值
    double pdfMean;//该波段均值
    double pdfStdDev;//该波段标准差

    //调用该函数后会自动生成影像名.aux.xml文件，用于记录统计好的信息
    if (err_Statistics == CE_Failure)
    {
        cout << "影像" << imgfile_ReadDir << "第" << BandIndex << "个波段的均值、标准差统计失败！" << endl;
    }
    else
    {
        cout << "影像" << imgfile_ReadDir << "第" << BandIndex << "个波段像素的均值Mean=" << pdfMean << "，标准差StdDev=" << pdfStdDev << endl;
    }


    //计算该波段的直方图
    //函数参数
    double dfMin = -0.5;//统计直方图的范围 ，假如是8bit影像，统计全部范围0-255则为(-0.5，255.5)(因为是开区间，所以范围要向外扩展)，
    double dfMax = 255.5;//也可以统计128-255,此时值设置为(127.5，255.5)
    int nBuckets = 256; //直方图统计份数，如果要把8bit影像的灰度范围0-255中的每个灰度为一份，则设置为256。
                         //假如0-1合并统计，2-3合并统计，则设置为128.

    memset(panHistogram, 0, sizeof(GUIntBig) * nBuckets);//将数组清零
    int bIncludeOutOfRange = TRUE;//是否统计区间外的值，有前两个参数可知有时统计的不是全部灰度范围

    int bApproxOK = FALSE;//是否进行粗略统计(粗略统计时精度降低速度加快)
    GDALProgressFunc pfnProgress = NULL;//表示进度条的回调函数
    void* pProgressData = NULL;//进度条数据
    //进行统计直方图
    CPLErr err_Histogram = poBand->GetHistogram(dfMin, dfMax, nBuckets, panHistogram, bIncludeOutOfRange, bApproxOK, pfnProgress, pProgressData);
    //调用该函数后会自动生成影像名.aux.xml文件，用于记录统计好的信息
    if (err_Histogram == CE_Failure)
    {
        cout << "影像" << imgfile_ReadDir << "第" << BandIndex << "个波段的直方图统计失败！" << endl;
    }


    //读取图像当前波段BandIndex的第一行数据
    int nXsize = poBand->GetXSize();
    float* pafScanline = new float[(size_t)nXsize];

    //关闭文件
    GDALClose((GDALDatasetH)poDataset);

void CreateRasterFile()
{
    //为了支持中文路径,请添加下面这句代码
    CPLSetConfigOption("GDAL_FILENAME_IS_UTF8", "NO");
    string pszRasterFile = "D:\\CreateTest.tif";

    //注册栅格驱动
    GDALAllRegister();

    //下面获取指定格式的驱动,用于创建图像
    string pszFormat = "GTiff";
    //GTiff创建为tif图像,HFA创建的为Erdas的img格式,ENVI创建为ENVI的hdr文件
    GDALDriver*poDriver=GetGDALDriverManager()->GetDriverByName(pszFormat.c_str());
    if (poDriver == NULL)
    {
        cout << "创建"<<pszRasterFile<<"文件时,获取格式" << pszFormat << "的驱动失败！" << endl;
        return;
    }

    //下面这几句是获取该驱动的一些元数据信息,里面有该格式是否支持创建等信息
    char **papszMetadata=poDriver->GetMetadata();
    if (CSLFetchBoolean(papszMetadata, GDAL_DCAP_CREATE, FALSE))
    {
        cout << "格式" << pszFormat << "支持Create()方法。" << endl;
    }
    if (CSLFetchBoolean(papszMetadata, GDAL_DCAP_CREATECOPY, FALSE))
    {
        cout << "格式" << pszFormat << "支持CreateCopy()方法。" << endl;
    }

    //创建输出文件,大小为*512,三个波段,8bit的数据
    GDALDataset* poDS = poDriver->Create(pszRasterFile.c_str(), 512, 512, 3, GDT_Byte, NULL); 
    if (poDS == NULL)
    {
        cout << "创建图像" << pszRasterFile << "失败！" << endl;
        return ;
    }

    ////设置投影
    //string pszWKT;//投影信息
    //poDS->SetProjection(pszWKT.c_str());//设置该图像的投影

    //设置地理变换参数
    //设置六参数,第一个和第四个是图像左上角的坐标,第二个和第六个为横向和纵向分辨率,余下两个为旋转角度
    double dGeotransform[6] = { 0,1,0,512,0,-1 }; 
    poDS->SetGeoTransform(dGeotransform);


    //关闭图像,此时图像里面全是0值,打开的话应该是黑色的
    GDALClose((GDALDatasetH)poDS); 

    cout << "图像" << pszRasterFile << "创建成功！已将其数据集关闭！" << endl;

    return;
}

3.更新影像

void UpdateRasterFile()
{
    //为了支持中文路径,请添加下面这句代码
    CPLSetConfigOption("GDAL_FILENAME_IS_UTF8", "NO");
    string pszRasterFile = "D:\\CreateTest.tif";

    //注册栅格驱动
    GDALAllRegister();

    //打开要更新的数据,注意第二个参数使用GA_Update 
    GDALDataset* poDS = (GDALDataset*)GDALOpen(pszRasterFile.c_str(), GA_Update);
    if (poDS == NULL)
    {
        cout << "影像" << pszRasterFile << "打开失败！" << endl;
        return;
    }

    //获取图像大小
    int iwidth = poDS->GetRasterXSize();
    int iHeight = poDS->GetRasterYSize();

    //声明一行图像大小的空间
    unsigned char* pBuf = new unsigned char[iwidth];
    memset(pBuf, 0, sizeof(unsigned char) * iwidth);

    //获取第一波段
    GDALRasterBand* pBand1 = poDS->GetRasterBand(1);
    //循环图像高,更新图像里面的像素值
    for (int i = 0; i < iHeight; i++)
    {
        //初始化当前行的像元值,像元值按照取余,图像应该是黑白条纹
        int iValue = i % 255;
        memset(pBuf, iValue, sizeof(unsigned char) * iwidth);
        //将像元值写入图像
        pBand1->RasterIO(GF_Write, 0, i, iwidth, 1, pBuf, iwidth, 1, GDT_Byte, 0, 0);
    }

    //获取第二波段
    GDALRasterBand* pBand2 = poDS->GetRasterBand(2);
    //将第二波段的值全部设置为128
    memset(pBuf, 128, sizeof(unsigned char) * iwidth);
    //循环图像高,更新图像里面的像素值
    for (int i = 0; i < iHeight; i++)
    {
        //将像元值写入图像
        pBand2->RasterIO(GF_Write, 0, i, iwidth, 1, pBuf, iwidth, 1, GDT_Byte, 0, 0);
    }

    //获取第三波段
    GDALRasterBand* pBand3 = poDS->GetRasterBand(3);
    //将第二波段的值全部设置为255
    memset(pBuf, 255, sizeof(unsigned char) * iwidth);
    //循环图像高,更新图像里面的像素值
    for (int i = 0; i < iHeight; i++)
    {
        //将像元值写入图像
        pBand3->RasterIO(GF_Write, 0, i, iwidth, 1, pBuf, iwidth, 1, GDT_Byte, 0, 0);
    }

    delete[]pBuf;
    //只有关闭数据集的时候,数据像元值等信息才会更新
    GDALClose((GDALDatasetH)poDS);
    cout << "影像" << pszRasterFile << "像元数据更新完成！" << endl;
}

函数调用示例：

    UpdateRasterFile();//更新影像

/**
    * @brief 删除栅格图像
    * @param pszFile    图像路径
    * @return 是否删除成功，成功为0
*/
int RasterDelete(const char* pszFile);//函数声明

int RasterDelete(const char* pszFile)
{
    GDALAllRegister();

    //打开图像
    GDALDataset* pDS = (GDALDataset*)GDALOpen(pszFile, GA_ReadOnly);
    if (pDS == NULL)
        return remove(pszFile);

    GDALDriver* pDriver = pDS->GetDriver();
    if (pDriver == NULL)
    {
        GDALClose((GDALDatasetH)pDS);
        return remove(pszFile);
    }

    GDALClose((GDALDatasetH)pDS);

    if (pDriver->Delete(pszFile) == CE_None)
        return 0;
    else
        return remove(pszFile);
}

/**
    * @brief 重命名/移动 栅格图像
    * @param pszOldFile            原始图像路径
    * @param pszNewFile            新图像路径
    * @return 是否删除成功，成功为0
*/
int RasterRename(const char* pszOldFile, const char* pszNewFile);//函数声明


int RasterRename(const char* pszOldFile, const char* pszNewFile)
{
    //如果要移动到某个文件夹中，这个文件夹必须存在
    GDALAllRegister();

    //打开图像
    GDALDataset* pDS = (GDALDataset*)GDALOpen(pszOldFile, GA_ReadOnly);
    if (pDS == NULL)
        return rename(pszOldFile, pszNewFile);

    GDALDriver* pDriver = pDS->GetDriver();
    if (pDriver == NULL)
    {
        GDALClose((GDALDatasetH)pDS);
        return remove(pszOldFile);
    }

    GDALClose((GDALDatasetH)pDS);

    if (pDriver->Rename(pszNewFile, pszOldFile) == CE_None)
        return 0;
    else
        return rename(pszOldFile, pszNewFile);
}

//分块读写影像
void BlockImageProcessing(CPLString InputImgPath, CPLString OutputImgPath)
{
    int BlockWidth = 512;//分块的列宽
    int BlockHeight = 512;//分块的行高
    GDALDataset* poDataset_Input; //GDAL数据集-输入 
    GDALDataset* poDataset_Output; //GDAL数据集-输出

    GDALAllRegister();
    poDataset_Input = (GDALDataset*)GDALOpen(InputImgPath, GA_ReadOnly);
    if (poDataset_Input == NULL)
    {
        cout << "文件打开失败！" << endl;
        return;
    }
    int BandCount = poDataset_Input->GetRasterCount();//输入影像波段数
    int nImgSizeX = poDataset_Input->GetRasterXSize();//输入影像宽
    int nImgSizeY = poDataset_Input->GetRasterYSize();//输入影像高
    GDALDataType datatype = poDataset_Input->GetRasterBand(1)->GetRasterDataType();//输入影像像元数据类型

    //创建新文件-输出影像
    CPLString format = "GTIFF";
    GDALDriver* poDriver = GetGDALDriverManager()->GetDriverByName(format);
    poDataset_Output = poDriver->Create(OutputImgPath, nImgSizeX, nImgSizeY, BandCount, datatype, NULL);
    if (poDataset_Output == NULL)
    {
        cout << "创建输出影像失败！" << endl;
        return;
    }

    double adfGeoTransform[6];
    poDataset_Input->GetGeoTransform(adfGeoTransform);//获取输入影像的仿射参数
    poDataset_Output->SetGeoTransform(adfGeoTransform);//设置输出影像的仿射参数
    poDataset_Output->SetProjection(poDataset_Input->GetProjectionRef());//设置投影

    //分块处理.将影像分成很多BlockWidth*BlockHeight大小的块，通过循环对每一块进行处理 
    int nxNum = (nImgSizeX - 1) / BlockWidth + 1;//计算列方向上分块块数 
    int nyNum = (nImgSizeY - 1) / BlockHeight + 1;//计算行方向上分块块数 
    int pafsizex; //当前块宽度 
    int pafsizey; //当前块高度 
    float* lp;
    float* ppafScan_Output = new float[(size_t)BlockWidth * BlockHeight * BandCount];//开辟内存
    memset(ppafScan_Output, 0, sizeof(float) * BlockWidth * BlockHeight * BandCount);//清零
    float* ppafScan_Input = new float[(size_t)BlockWidth * BlockHeight * BandCount];//开辟内存
    memset(ppafScan_Input, 0, sizeof(float) * BlockWidth * BlockHeight * BandCount);//清零
    for (int nYI = 0; nYI < nyNum; nYI++)
        for (int nXI = 0; nXI < nxNum; nXI++)
        {
            //依次处理每个分块
            pafsizex = BlockWidth;
            pafsizey = BlockHeight;
            //行列末尾余数的小块的处理 
            if (nXI == nxNum - 1)pafsizex = (nImgSizeX - 1) % BlockWidth + 1;
            if (nYI == nyNum - 1)pafsizey = (nImgSizeY - 1) % BlockHeight + 1;

            GDALDataset* poDataset_Output_block; //GDAL数据集-输出
            GDALDriver* poDriver_block = GetGDALDriverManager()->GetDriverByName(format);
            string OutputImgPath_block = OutputImgPath.substr(0, OutputImgPath.rfind('.'))  + "_block" + to_string(nYI * nxNum + nXI + 1) + OutputImgPath.substr(OutputImgPath.rfind('.'));
            poDataset_Output_block = poDriver_block->Create(OutputImgPath_block.c_str(), pafsizex, pafsizey, BandCount, datatype, NULL);
            if (poDataset_Output_block == NULL)
            {
                cout << "创建输出影像分块" << nYI * nxNum + nXI + 1 << "失败！" << endl;
                return;
            }
            double ZeroX = adfGeoTransform[0];//影像左上角顶点的列坐标（经度）
            double ZeroY = adfGeoTransform[3];//影像左上角顶点的行坐标（纬度）
            double PixelX = adfGeoTransform[1];//影像分辨率（像元宽）
            double PixelY = adfGeoTransform[5];//影像分辨率（像元高）备注：PixelY是矢量，因Y轴向下为正方向，所以为负数
            //像素行列坐标 求 地理坐标    col:列数    row:行数
            int row = nYI * BlockHeight;
            int col = nXI * BlockWidth;
            double GeoX = ZeroX + col * PixelX + row * adfGeoTransform[2];//地理列坐标（经度）
            double GeoY = ZeroY + col * adfGeoTransform[4] + row * PixelY;//地理行坐标（纬度）

            double adfGeoTransform_block[6] = {GeoX,PixelX,adfGeoTransform[2], GeoY ,adfGeoTransform[4],PixelY};//0和3是左上角顶点的xy坐标。1和5是像元的宽高
            //输出当前分块信息
            cout << "当前分块" << nYI * nxNum + nXI + 1 << "坐标范围：" << endl;
            printf("左上角origin=(%.6f,%.6f)\n", adfGeoTransform_block[0], adfGeoTransform_block[3]);
            printf("右下角end=(%.6f,%.6f)\n", adfGeoTransform_block[0] + (pafsizex-1) * adfGeoTransform_block[1] + (pafsizey-1) * adfGeoTransform_block[2], adfGeoTransform_block[3] + (pafsizex-1) * adfGeoTransform_block[4] + (pafsizey-1)  * adfGeoTransform_block[5]);
            printf("像元尺寸大小矢量PixelSize=(%.6f,%.6f)\n\n", adfGeoTransform_block[1], adfGeoTransform_block[5]);
            poDataset_Output_block->SetGeoTransform(adfGeoTransform_block);//设置输出影像的仿射参数
            poDataset_Output_block->SetProjection(poDataset_Input->GetProjectionRef());//设置投影
            //float* ppafScan_Output_block = new float[(size_t)pafsizex * pafsizey * BandCount];//开辟内存
            //memset(ppafScan_Output_block, 0, sizeof(float) * pafsizex * pafsizey * BandCount);//清零
            //读取当前块数据 
            poDataset_Input->RasterIO(GF_Read, nXI * BlockWidth, nYI * BlockHeight, pafsizex, pafsizey, ppafScan_Input, pafsizex, pafsizey, datatype, BandCount, 0, BandCount * sizeof(float), pafsizex * BandCount * sizeof(float), sizeof(float));
            //对当前块进行处理
            //顺序 波段 行 列
            /*
            for (int nnum = 0; nnum < BandCount; nnum++)//波段
            {
                for (int i = 0; i < pafsizey; i++)//行
                {
                    for (int j = 0; j < pafsizex; j++)//列
                    {
                        lp = ppafScan_Input + nnum * pafsizex * pafsizey + i * pafsizex + j;
                        *(ppafScan_Output + nnum * pafsizex * pafsizey + i * pafsizex + j) = *lp;
                    }
                }
            }
            */
            for (int i = 0; i < pafsizey; i++)//行
            {
                for (int j = 0; j < pafsizex; j++)//列
                {
                    for (int nnum = 0; nnum < BandCount; nnum++)//波段
                    {
                        lp = ppafScan_Input + i * pafsizex * BandCount + j * BandCount + nnum;
                        *(ppafScan_Output + i * pafsizex * BandCount + j * BandCount + nnum) = *lp;
                        //*(ppafScan_Output_block + i * pafsizex * BandCount + j * BandCount + nnum) = *lp;
                    }
                }
            }
            //将当前块数据写入新图像相应位置 
            poDataset_Output->RasterIO(GF_Write, nXI * BlockWidth, nYI * BlockHeight, pafsizex, pafsizey, ppafScan_Output, pafsizex, pafsizey, datatype, BandCount, 0, BandCount * sizeof(float), pafsizex * BandCount * sizeof(float), sizeof(float));
            //poDataset_Output->RasterIO(GF_Write, nXI * BlockWidth, nYI * BlockHeight, pafsizex, pafsizey, ppafScan_Output, pafsizex, pafsizey, datatype, BandCount, 0, 0, 0, 0);
            //输出保存当前分块影像ppafScan_Output_block
            poDataset_Output_block->RasterIO(GF_Write, 0, 0, pafsizex, pafsizey,ppafScan_Output , pafsizex, pafsizey, datatype, BandCount, 0, BandCount * sizeof(float), pafsizex * BandCount * sizeof(float), sizeof(float));
            //delete[] ppafScan_Output_block;
            GDALClose((GDALDatasetH)poDataset_Output_block);
        }
    delete[] ppafScan_Input;
    delete[] ppafScan_Output;
    //写操作完成后必须释放，不然写入操作不成功 
    GDALClose((GDALDatasetH)poDataset_Input);
    GDALClose((GDALDatasetH)poDataset_Output);

    cout << endl << "影像" << OutputImgPath << "创建成功！" << endl;
    return;
}
