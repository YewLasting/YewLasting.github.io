1269. 停在原地的方案数

难度：困难

题干

有一个长度为 arrLen 的数组，开始有一个指针在索引 0 处。

每一步操作中，你可以将指针向左或向右移动 1 步，或者停在原地（指针不能被移动到数组范围外）。

给你两个整数 steps 和 arrLen ，请你计算并返回：在恰好执行 steps 次操作以后，指针仍然指向索引 0 处的方案数。

由于答案可能会很大，请返回方案数 模 10^9 + 7 后的结果。

示例 1：

输入：steps = 3, arrLen = 2
输出：4
解释：3 步后，总共有 4 种不同的方法可以停在索引 0 处。
向右，向左，不动
不动，向右，向左
向右，不动，向左
不动，不动，不动

示例 2：

输入：steps = 2, arrLen = 4
输出：2
解释：2 步后，总共有 2 种不同的方法可以停在索引 0 处。
向右，向左
不动，不动

示例 3：

输入：steps = 4, arrLen = 2
输出：8

提示：

  • 1 <= steps <= 500
  • 1 <= arrLen <= 10^6

来源：力扣（LeetCode）

链接：https://leetcode-cn.com/problems/number-of-ways-to-stay-in-the-same-place-after-some-steps/

题解

语言：

C++

#include <iostream>
#include <vector>
#include <queue>
#include <set>
#include <stack>
#include <cstring>
#include <algorithm>
#include <unordered_map>
#include <unordered_set>
#include <numeric>
using namespace std;
class Solution
{
public:
    int numWays(int steps, int arrLen)
    {
        //dp[i][j]代表经过i步后回到下标j的方案数量
        //考虑dp数组的范围
        //0<=i<=steps
        //0<=j<=arrLen-1
        //因为假设每一步都是向右移动，所以经过steps后抵达的下标位置最远为steps,所以j的范围可进一步缩小
        //0<=j<=min(arrLen-1,steps)
        int cols = min(arrLen - 1, steps);
        //vector<vector<int>> dp(steps+1,vector<int>(cols+1,0));//初始值全部设置为0
        //考虑动态规划的初始值
        //当steps=0时，指针初始时位于索引0处，因此dp[0][0]=1,可以视为此时只有一种方案
        //而dp[0][1~cols]=0，视为0种方案
        //dp[0][0]=1;
        //考虑状态转移方程
        //           正上方左边方案数  正上方方案数   正上方右边方案数
        //dp[i][j]=  dp[i-1][j-1]   + dp[i-1][j]  +  dp[i-1][j+1];//当前的索引可能来自于上一个索引的右移,不动,左移。因此当前的方案数来自于以上三种情况的方案数之和
        //最后dp[steps][0]即为所求的经过steps步后，指针仍然指向索引0时的方案数

        //注意：方案数可能非常大，题目要求对结果取模，因此每次计算dp[i][j]后还要进行取模运算
        int mod = 1000000007;
        //dp[i][j]=dp[i][j]%mod;

        //此外考虑到指针不可以移动到数组范围外，当j=0时，正上方左边方案数索引越界了,dp[i-1][j-1]视为0进行计算
        //当j=clos时，正上方右边方案数索引越界了,dp[i-1][j+1]视为0进行计算

        //考虑滚动数组优化
        //考虑到当前状态只和上一行状态有关系，因此可以开辟一行滚动数组，无需开辟整个动态规划矩阵
        vector<int> dp(cols + 1, 0);
        dp[0] = 1;

        for (int i = 1; i <= steps; ++i)
        {
            vector<int> dp_now(cols + 1, 0);
            for (int j = 0; j <= cols; ++j)
            {
                // if(j==0)
                // {
                //     dp_now[j]=  (dp[j]  +  dp[j+1])%mod;
                // }
                // else if(j==cols)
                // {
                //     dp_now[j]=  (dp[j-1]   + dp[j]) %mod;
                // }
                // else
                // {
                //     dp_now[j]=  (dp[j-1]   + dp[j] )%mod;
                //     dp_now[j]=  (dp_now[j] +  dp[j+1])%mod;//分两步加，然后取两次余//否则会超出数据类型的范围
                // }

                //法二：
                dp_now[j] = dp[j];
                if (j > 0)
                {
                    dp_now[j] = (dp_now[j] + dp[j - 1]) % mod;
                }
                if (j < cols)
                {
                    dp_now[j] = (dp_now[j] + dp[j + 1]) % mod;
                }
            }
            dp = dp_now;
        }
        return dp[0];
    }
};
//进行测试，输出测试结果
int main()
{
    Solution test;
    int steps = 4;
    int arrLen = 2;
    int res = test.numWays(steps, arrLen);
    cout << res << endl;
    getchar();
    return 0;
}
