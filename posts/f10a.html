〇、STL基础

STL基础知识

一、vector介绍

数据结构：向量(vector)

说明：连续存储元素。底层数据结构为数组，支持快速随机访问

实现头文件：<vector>

  • 是一个向量类模板，相当于数组。
  • 用于存储具有相同数据类型的一组元素，可以从末尾快速的插入与删除元素，快速地随机访问元素。
  • 但是在序列中间插入、删除元素较慢，因为需要移动插入或删除处后面的所有元素。

1.初始化方式

vector<int> v1;                     //定义元素为int的向量v1
vector<int> v2(10);                 //指定向量v2的初始大小为10个int元素
vector<double> v3(10，1.23);        //指定v3的10个初始元素的初值为1.23
vector<int> v4(a，a+5);            //用数组a[0..4]共5个元素初始化v4

vector<int> a
a.assign(b.begin(), b.begin()+3);     //b为向量，将b的0~2个元素构成的向量赋给a
a.assign(4,2);                         //a只含4个元素，且每个元素为2

2.主要成员函数

[]                //返回指定下标的元素。
front()            //获取当前向量容器的第一个元素。
back()            //获取当前向量容器的最后一个元素。

size()            //返回当前向量容器的中的实际元素个数。

pop_back()         //删除向量的最后一个元素
push_back()        //在当前向量容器尾部添加了一个元素。
push_back(5);     //在向量的最后一个元素后插入一个元素，其值为5

insert(pos，elem)//在pos位置插入元素elem，即将元素elem插入到迭代器pos指定元素之前。
a.insert(a.begin()+1,5);     //在a的第1个元素（从第0个算起）的位置插入数值5，如a为1,2,3,4，插入元素后为1,5,2,3,4
a.insert(a.begin()+1,3,5);  //在a的第1个元素（从第0个算起）的位置插入3个数，其值都为5
a.insert(a.begin()+1,b+3,b+6);  //b为数组，在a的第1个元素（从第0个算起）的位置插入b的第3个元素到第5个元素（不包括b+6）
                                //  如b为1,2,3,4,5,9,8         
                                // 插入元素后为1,4,5,9,2,3,4,5,9,8

reserve(n)        //为当前向量容器预分配n个元素的存储空间。
capacity()        //返回当前向量容器在重新进行内存分配以前所能容纳的元素个数。

resize(n)        //调整当前向量容器的大小，使其能容纳n个元素，多则删，少则补，其值随机
resize(n,2);     //调整当前向量容器的大小，使其能容纳n个元素，多则删，少则补，其值为2
reserve(100);     //当前向量容器的（capacity）扩充至100，也就是说现在测试a.capacity();的时候返回值是100.
                //这种操作只有在需要给a添加大量数据的时候才显得有意义，因为这将避免内存多次容量扩充操作
                //（当a的容量不足时电脑会自动扩容，当然这必然降低性能）    

erase()            //删除当前向量容器中某个迭代器或者迭代器区间指定的元素。
a.erase(a.begin()+1,a.begin()+3); //删除a中第1个（从第0个算起）到第2个元素
                                  //即删除a[1],a[2]这两个元素，a[0],a[3]没有被删除
                                  //也就是说删除的元素从a.begin()+1算起（包括它）一直到a.begin()+3（不包括它）    

clear()            //删除当前向量容器中所有元素。
empty()            //判断当前向量容器是否为空，空则返回ture,不空则返回false

a.swap(b);         //b为向量，将a中的元素和b中的元素进行整体性交换
a==b;             //b为向量，向量的比较操作还有!=,>=,<=,>,<

//迭代器函数：
begin()、end()、rbegin()、rend()

代码示例：

#include <vector>
using namespace std;
void main()
{   
    vector<int> myv;            //定义vector容器myv
    vector<int>::iterator it;    //定义myv的正向迭代器it 备注反向迭代器为：reverse_iterator
    myv.push_back(1);            //在myv末尾添加元素1
    it=myv.begin();                //it迭代器指向开头元素1
    myv.insert(it,2);            //在it指向的元素之前插入元素2
    myv.push_back(3);            //在myv末尾添加元素3
    myv.push_back(4);            //在myv末尾添加元素4
    it=myv.end();            //it迭代器指向尾元素4的后面
    it--;                    //it迭代器指向尾元素4
    myv.erase(it);            //删除元素4
    for (it=myv.begin();it!=myv.end();++it)
    {
        printf("%d ",*it);
    }
    printf("\n");
}

3.给vector添加元素

(1)直接添加元素

vector<int> a;
for(int i=0;i<10;i++)
    a.push_back(i);

(2)从数组中取值添加元素

int a[6]={1,2,3,4,5,6};
vector<int> b；
for(int i=1;i<=4;i++)
    b.push_back(a[i]);//2,3,4,5

(3)从已有向量中取值添加元素

int a[6]={1,2,3,4,5,6};
vector<int> b;
vector<int> c(a,a+4);//1,2,3,4
vector<int>::iterator it//定义c的正向迭代器it
for(it=c.begin();it!=c.end();++it)
        b.push_back(*it);

(4)从文件中取值添加元素

ifstream in("data.txt");
vector<int> a;
for(int i; in>>i)
    a.push_back(i);

(5)错误做法

vector<int> a;
for(int i=0;i<10;i++)
    a[i]=i;

理由：下标只能用于获取已存在的元素，而现在的a[i]还是空的对象

4.访问vector中的元素

(1)通过下标访问

int a[6]={1,2,3,4,5,6};
vector<int> b(a,a+4);
for(int i=0;i<b.size();++i)
    cout<<b[i]<<" ";

(2)通过迭代器访问

int a[6]={1,2,3,4,5,6};
vector<int> b(a,a+4);
vector<int>::iterator it//定义b的正向迭代器it
for(it=b.begin();it!=b.end();++it)
    cout<<*it<<" ";

5.常用算法

使用时要包含头文件

#include<algorithm>

(1)sort排序

sort(a.begin(),a.end()); //对a中的从a.begin()（包括它）到a.end()（不包括它）的元素进行从小到大排列

sort(a.begin(),a.end(), less<int>()); //升序
sort(a.begin(),a.end(), greater<int>()); //降序

vector<int> data = { 1, 3, 2, 4 };

bool upsort(const int &a, const int &b)
{
    return a < b;
}
sort(data.begin(), data.end(), upsort);   //升序

bool downsort(const int &a, const int &b)
{
    return a > b;
}
sort(data.begin(), data.end(), downsort); //降序
//is_sorted函数检测两个迭代器指定范围是否是已经按照升序排列
is_sorted(A.begin(), A.end()) //如果A中元素升序，则返回true
is_sorted(A.rbegin(), A.rend());//如果A中元素降序，则返回true

(2)reverse倒置

reverse(a.begin(),a.end()); //对a中的从a.begin()（包括它）到a.end()（不包括它）的元素倒置，但不排列，
                            //如a中元素为1,3,2,4,倒置后为4,2,3,1

(3)copy复制

copy(a.begin(),a.end(),b.begin()+1); //把a中的从a.begin()（包括它）到a.end()（不包括它）的元素复制到b中，
                                     //从b.begin()+1的位置（包括它）开始复制，覆盖掉原有元素

(4)find查找

find(a.begin(),a.end(),10); //在a中的从a.begin()（包括它）到a.end()（不包括它）的元素中查找10，
                            //若存在返回其在向量中的位置 返回的是迭代器

(5)最大值、最小值

auto data_max = data.max_element(data.begin(), data.end());    //返回的是迭代器

auto data_min = data.min_element(data.begin(), data.end());

(6)求和

使用时要包含头文件

#include <numeric>
int sum = accumulate(data.begin(), data.end(), 0);
int sum = accumulate(data.begin(), data.end(), (int)0);

double sum = accumulate(data.begin(), data.end(), 0.0);           //应避免精度损失

(7)合并

vector<int> vec1 = {...};

vector<int> vec3;//vec3是空的
vec3.insert(vec3.end(),vec1.begin(),vec1.end())//将vec1压入
vec3.insert(vec3.end(),vec2.begin(),vec2.end())//继续将vec2压入

6.C++11新特性

C++11中新加入了emplace_back()，作用与push_back()类似，不过效率更高，理由如下：

如果要将一个临时变量push到容器的末尾，push_back()需要先构造临时对象，再将这个对象拷贝到容器的末尾，而emplace_back()则直接在容器的末尾构造对象，这样就省去了拷贝的过程。

a.emplace_back(i) //将i添加到a的末尾
